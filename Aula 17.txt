CREATE TABLE ALUNO (
    id_aluno      INT          NOT NULL,
    nome          VARCHAR(100) NOT NULL,
    cpf           CHAR(11)     NOT NULL,
    dt_nascimento DATE         NOT NULL,
    endereco      VARCHAR(200) NOT NULL,
    telefone      VARCHAR(15)  NOT NULL,
    CONSTRAINT PK_ALUNO     PRIMARY KEY (id_aluno),
    CONSTRAINT UK_ALUNO_CPF UNIQUE      (cpf)
);

CREATE TABLE PROFESSOR (
    id_professor INT          NOT NULL,
    nome         VARCHAR(100) NOT NULL,
    CONSTRAINT PK_PROFESSOR PRIMARY KEY (id_professor)
);


CREATE TABLE CURSO (
    id_curso   INT          NOT NULL,
    nome_curso VARCHAR(100) NOT NULL,
    CONSTRAINT PK_CURSO PRIMARY KEY (id_curso)
);


CREATE TABLE OFERTA_CURSO (
    id_oferta_curso INT  NOT NULL,
    id_curso        INT  NOT NULL,
    ano             date NOT NULL,
    CONSTRAINT PK_OFERTA_CURSO    PRIMARY KEY (id_oferta_curso),
    CONSTRAINT FK_OFERTA_CURSO_CU FOREIGN KEY (id_curso)
        REFERENCES CURSO (id_curso),
    CONSTRAINT UK_OFERTA_CURSO    UNIQUE (id_curso, ano)
);


CREATE TABLE DISCIPLINA (
    id_disciplina   INT          NOT NULL,
    nome_disciplina VARCHAR(100) NOT NULL,
    carga_horaria   SMALLINT     NOT NULL CHECK (carga_horaria > 0),
    CONSTRAINT PK_DISCIPLINA PRIMARY KEY (id_disciplina)
);


CREATE TABLE OFERTA_DISCIPLINA (
    id_oferta_disc  INT     NOT NULL,
    id_disciplina   INT     NOT NULL,
    id_oferta_curso INT     NOT NULL,
    id_professor    INT     NOT NULL,
    semestre        int NOT NULL,
    CONSTRAINT PK_OFERTA_DISC     PRIMARY KEY (id_oferta_disc),
    CONSTRAINT FK_OD_DISCIPLINA   FOREIGN KEY (id_disciplina)
        REFERENCES DISCIPLINA   (id_disciplina),
    CONSTRAINT FK_OD_OFERTA_CURSO FOREIGN KEY (id_oferta_curso)
        REFERENCES OFERTA_CURSO (id_oferta_curso),
    CONSTRAINT FK_OD_PROFESSOR    FOREIGN KEY (id_professor)
        REFERENCES PROFESSOR    (id_professor),
    CONSTRAINT CHK_SEMESTRE       CHECK (semestre IN (1, 2)),
    CONSTRAINT UK_OFERTA_DISC     UNIQUE (id_disciplina, id_oferta_curso, semestre)
);


CREATE TABLE MATRICULA (
    id_matricula   INT  NOT NULL,
    id_aluno       INT  NOT NULL,
    id_oferta_disc INT  NOT NULL,
    dt_matricula   DATE NOT NULL DEFAULT (CURRENT_DATE),
    CONSTRAINT PK_MATRICULA       PRIMARY KEY (id_matricula),
    CONSTRAINT FK_MAT_ALUNO       FOREIGN KEY (id_aluno)
        REFERENCES ALUNO             (id_aluno),
    CONSTRAINT FK_MAT_OFERTA_DISC FOREIGN KEY (id_oferta_disc)
        REFERENCES OFERTA_DISCIPLINA (id_oferta_disc),
    CONSTRAINT UK_MATRICULA       UNIQUE (id_aluno, id_oferta_disc)
);

FROM OFERTA_CURSO
INNER JOIN CURSO ON CURSO.id_curso = OFERTA_CURSO.id_curso;

SELECT nome_curso, ano, semestre, nome_disciplina, nome
FROM OFERTA_DISCIPLINA
INNER JOIN OFERTA_CURSO ON OFERTA_CURSO.id_oferta_curso = OFERTA_DISCIPLINA.id_oferta_curso
INNER JOIN CURSO        ON CURSO.id_curso               = OFERTA_CURSO.id_curso
INNER JOIN DISCIPLINA   ON DISCIPLINA.id_disciplina     = OFERTA_DISCIPLINA.id_disciplina
